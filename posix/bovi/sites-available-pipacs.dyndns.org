<VirtualHost *:443>

DocumentRoot "/var/www/html"
ServerName pipacs.dyndns.org
LogLevel warn
SSLEngine on
SSLProtocol all -SSLv2
SSLCipherSuite ALL:!ADH:!EXPORT:!SSLv2:RC4+RSA:+HIGH:+MEDIUM:+LOW
SSLCertificateFile /etc/pki/tls/certs/pipacs.dyndns.org.cert
SSLCertificateKeyFile /etc/pki/tls/private/pipacs.dyndns.org.key

<Files ~ "\.(cgi|shtml|phtml|php3?)$">
    SSLOptions +StdEnvVars
</Files>

<Directory "/var/www/cgi-bin">
    SSLOptions +StdEnvVars
</Directory>

SetEnvIf User-Agent ".*MSIE.*" \
         nokeepalive ssl-unclean-shutdown \
         downgrade-1.0 force-response-1.0

<Directory "/var/www/html">
    Options Indexes FollowSymLinks
    AllowOverride None
    SSLRequireSSL
    AuthType basic
    AuthName "Pipacs"
    AuthUserFile /var/www/conf/passwd
    Require valid-user
</Directory>

<Directory "/var/www/html/dav">
    DAV On
    Options FollowSymLinks Indexes MultiViews
    AllowOverride None
    AuthType basic
    AuthName "Music"
    AuthUserFile /var/www/conf/passwd
    Require valid-user
</Directory>

</VirtualHost>
